Custom pagination template

Page: {[tableParams.page]}

Count per page: {[tableParams.count]}
{[user.name]} {[user.age]}

Html

<table ng-table="tableParams" template-pagination="custom/pager" class="table">
    <tr ng-repeat="user in users">
        <td title="Name">
            {[user.name]}
        </td>
        <td title="Age">
            {[user.age]}
        </td>
    </tr>
</table>
<!-- Custom template -->
<script type="text/ng-template" id="custom/pager">
<ul class="pager ng-cloak">
<li ng-repeat="page in pages"
     ng-class="{'disabled': !page.active, 'previous': page.type == 'prev', 'next': page.type == 'next'}"
     ng-show="page.type == 'prev' || page.type == 'next'" ng-switch="page.type">

    <a ng-switch-when="prev" ng-click="goToPage(page.number)" href="">« Previous</a>

    <a ng-switch-when="next" ng-click="goToPage(page.number)" href="">Next »</a>
</li>
<li>
    <!-- Count per page -->
    <div class="btn-group">
        <button type="button"
            ng-class="{'active':params.count == 10}"
            ng-click="changeCount(10)" class="btn">10</button>
        <button type="button"
            ng-class="{'active':params.count == 25}"
            ng-click="changeCount(25)" class="btn">25</button>
        <button type="button"
            ng-class="{'active':params.count == 50}"
            ng-click="changeCount(50)" class="btn">50</button>
        <button type="button"
            ng-class="{'active':params.count == 100}"
            ng-click="changeCount(100)" class="btn">100</button>
    </div>
</li>
</ul>
</script>

Javascript

function DemoCtrl($scope, ngTableParams) {
    var data = [{name: "Moroni", age: 50},
                // more data...
                {name: "Enos", age: 34}];

    $scope.tableParams = new ngTableParams({
        page: 1,            // show first page
        total: data.length, // length of data
        count: 10           // count per page
    });

    // watch for changes of parameters
    $scope.$watch('tableParams', function(params) {
        // slice array data on pages
        $scope.users = data.slice(
            (params.page - 1) * params.count,
            params.page * params.count
        );
    }, true);
}
