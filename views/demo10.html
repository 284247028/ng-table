Html

<table ng-table="tableParams" show-filter="true" class="table">
    <tr ng-repeat="user in users" ng-class="{ 'emphasis': user.money > 500 }">
        <td title="" filter="{ 'checked': 'checkbox' }">
            <input type="checkbox" ng-model="user.checked" />
        </td>
        <td title="Name" filter="{ 'name': 'select' }" filter-data="names($column)">
            {{user.name}}
        </td>
        <td title="Money" filter="{ 'action': 'button' }" sortable="money">
            {{user.money}}
        </td>
    </tr>
</table>
<script type="text/ng-template" id="ng-table/filters/checkbox.html">
    <input type="checkbox" ng-model="checkboxes.checked" id="select_all" name="filter-checkbox" value="" />
</script>

Javascript:

// IMPORTANT! Must be an object
$scope.checkboxes = { 'checked': false };

// watch for check all checkbox
$scope.$watch('checkboxes.checked', function(value) {
    angular.forEach($scope.users, function(item) {
        item.checked = value;
    });
});

// watch for data checkboxes
$scope.$watch('users', function(values) {
    var checked = 0, unchecked = 0;
    angular.forEach(values, function(item) {
        checked   +=  (item.checked) || 0;
        unchecked += (!item.checked) || 0;
    });
    // grayed checkbox
    $("#select_all").prop("indeterminate", (checked != 0 && unchecked != 0));
    $scope.checkboxes.checked = (unchecked == 0) && (checked > 0);
}, true);

$scope.$watch('tableParams', function(params) {
    // use build-in angular filter
    var orderedData = params.sorting ? 
                        $filter('orderBy')(data, params.orderBy()) :
                        data;
    orderedData = params.filter ? 
                        $filter('filter')(orderedData, params.filter) :
                        orderedData;

    params.total = orderedData.length; // set total for recalc pagination
    $scope.users = orderedData.slice((params.page - 1) * params.count, params.page * params.count);
}, true);
